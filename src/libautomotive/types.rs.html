1//! Common types used throughout the automotive protocol stack.
2//!
3//! This module provides fundamental types and traits that are used across different
4//! layers of the protocol stack. It includes basic types for CAN communication,
5//! addressing, and frame structures, as well as traits for configuration and
6//! hardware port interactions.
7
8/// CAN identifier type, supporting both standard (11-bit) and extended (29-bit) identifiers.
9pub type CanId = u32;
10
11/// Frame data type representing the payload of a CAN frame.
12///
13/// The maximum length depends on the protocol:
14/// - Classic CAN: 8 bytes
15/// - CAN-FD: up to 64 bytes
16pub type FrameData = Vec<u8>;
17
18/// Timestamp type representing milliseconds since an arbitrary epoch.
19///
20/// Used for timing and synchronization purposes across the protocol stack.
21pub type Timestamp = u64;
22
23/// Protocol-specific addressing information, primarily used in higher layer protocols
24/// like J1939.
25#[derive(Debug, Clone, Copy, PartialEq, Eq)]
26pub struct Address {
27    /// Message priority (0-7, with 0 being highest priority)
28    pub priority: u8,
29    /// Parameter Group Number (PGN) identifying the message type
30    pub pgn: u32,
31    /// Source address of the sending node
32    pub source: u8,
33    /// Destination address of the target node
34    pub destination: u8,
35}
36
37/// Generic frame structure used across different protocol layers.
38///
39/// This structure provides a unified representation of CAN frames,
40/// supporting both classic CAN and CAN-FD formats.
41#[derive(Debug, Clone)]
42pub struct Frame {
43    /// CAN identifier (11-bit or 29-bit)
44    pub id: CanId,
45    /// Frame payload data
46    pub data: FrameData,
47    /// Timestamp of frame reception/transmission
48    pub timestamp: Timestamp,
49    /// Whether the frame uses extended (29-bit) identifier
50    pub is_extended: bool,
51    /// Whether the frame is a CAN-FD frame
52    pub is_fd: bool,
53}
54
55impl Default for Frame {
56    fn default() -> Self {
57        Self {
58            id: 0,
59            data: Vec::new(),
60            timestamp: 0,
61            is_extended: false,
62            is_fd: false,
63        }
64    }
65}
66
67/// Configuration trait that must be implemented by all protocol configurations.
68///
69/// This trait ensures that protocol configurations can be validated before use
70/// and can be safely shared between threads.
71pub trait Config: Send + Sync {
72    /// Validates the configuration parameters.
73    ///
74    /// Returns `Ok(())` if the configuration is valid, or an appropriate error
75    /// if validation fails.
76    fn validate(&self) -> crate::error::Result<()>;
77}
78
79/// Hardware abstraction trait for CAN interfaces.
80///
81/// This trait must be implemented by platform-specific code to provide
82/// the actual hardware communication capabilities.
83pub trait Port: Send + Sync {
84    /// Sends a frame through the CAN interface.
85    fn send(&mut self, frame: &Frame) -> crate::error::Result<()>;
86
87    /// Receives a frame from the CAN interface.
88    ///
89    /// This method will block until a frame is received or a timeout occurs.
90    fn receive(&mut self) -> crate::error::Result<Frame>;
91
92    /// Sets the timeout for receive operations.
93    ///
94    /// # Parameters
95    /// * `timeout_ms` - Timeout in milliseconds. A value of 0 means no timeout.
96    fn set_timeout(&mut self, timeout_ms: u32) -> crate::error::Result<()>;
97}
