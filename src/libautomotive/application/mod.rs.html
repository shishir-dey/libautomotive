1//! Application layer implementations for automotive protocols.
2//!
3//! This module provides implementations for application layer protocols:
4//! - UDS (ISO 14229-1) - Unified Diagnostic Services
5//! - OBD-II (ISO 15031) - On-Board Diagnostics
6//!
7//! The application layer provides high-level diagnostic and monitoring services:
8//!
9//! ## UDS (Unified Diagnostic Services)
10//! - ECU programming and configuration
11//! - Diagnostic trouble code (DTC) management
12//! - Data reading and writing
13//! - Routine control
14//! - Security access
15//!
16//! ## OBD-II (On-Board Diagnostics)
17//! - Emissions-related diagnostics
18//! - Real-time parameter monitoring
19//! - Freeze frame data access
20//! - Standard diagnostic trouble codes
21//! - Mode-based service requests
22//!
23//! # Examples
24//!
25//! ```text
26//! # Example usage of the application layer (conceptual, not actual code)
27//! # UDS configuration:
28//! let uds_config = uds::CustomConfig {
29//!     timeout_ms: 1000,             // 1 second timeout
30//!     p2_timeout_ms: 5000,          // 5 second P2 timeout
31//!     tester_present_interval_ms: 2000  // Send tester present every 2 seconds
32//! };
33//!
34//! # Create a UDS instance
35//! let mut uds = uds::CustomInterface::new(uds_config);
36//! uds.open();
37//!
38//! # Change diagnostic session
39//! uds.change_session(uds::SESSION_PROGRAMMING);
40//!
41//! # Read ECU data
42//! let vin_data = uds.read_data_by_id(0xF190); // Vehicle Identification Number
43//!
44//! # Write ECU data
45//! uds.write_data_by_id(0xF198, &[0x01, 0x02, 0x03]);
46//!
47//! # OBD-II configuration:
48//! let obd_config = obdii::CustomConfig {
49//!     timeout_ms: 1000,    // 1 second timeout
50//!     auto_format: true    // Auto-format responses
51//! };
52//!
53//! # Create an OBD-II instance
54//! let mut obd = obdii::CustomInterface::new(obd_config);
55//! obd.open();
56//!
57//! # Read engine RPM
58//! let engine_rpm = obd.read_sensor(0x0C); // PID for engine RPM
59//!
60//! # Read diagnostic trouble codes
61//! let dtcs = obd.read_dtc();
62//! ```

73/// Application layer trait that must be implemented by UDS and OBD-II
74pub trait ApplicationLayer {
75    type Config: Config;
76    type Request;
77    type Response;
78
79    fn new(config: Self::Config) -> Result<Self>
80    where
81        Self: Sized;
82    fn open(&mut self) -> Result<()>;
83    fn close(&mut self) -> Result<()>;
84    fn send_request(&mut self, request: &Self::Request) -> Result<Self::Response>;
85    fn set_timeout(&mut self, timeout_ms: u32) -> Result<()>;
86}
