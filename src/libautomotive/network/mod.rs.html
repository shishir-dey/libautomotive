1//! Network layer implementations for automotive protocols.
2//!
3//! This module provides implementations for network layer protocols, primarily:
4//! - J1939 (SAE J1939) - A higher-layer protocol for commercial vehicles
5//!
6//! The network layer is responsible for:
7//! - Message routing and addressing
8//! - Network management
9//! - Message prioritization
10//! - Protocol-specific addressing schemes
11//!
12//! The J1939 protocol is widely used in commercial vehicles and provides:
13//! - Parameter Group Numbers (PGN) based addressing
14//! - Multi-packet message transport
15//! - Network management functions
16//! - Standardized diagnostic messages
17//!
18//! # Examples
19//!
20//! ```text
21//! # Example usage of the network layer (conceptual, not actual code)
22//! # J1939 configuration:
23//! let j1939_config = j1939::CustomConfig {
24//!     name: 0x0000AABBCCDDEEFF, // ECU NAME (64-bit identifier)
25//!     address: 0x42,            // Preferred source address
26//!     priority: 6               // Default priority for messages
27//! };
28//!
29//! # Create a J1939 instance and open the connection
30//! let mut j1939 = j1939::CustomInterface::new(j1939_config);
31//! j1939.open();
32//!
33//! # Claim a network address
34//! j1939.claim_address(0x42);
35//!
36//! # Send a J1939 message
37//! let pgn = 0x00EF00;     // Example PGN (Electronic Engine Controller)
38//! let dest_addr = 0xFF;   // Broadcast address
39//! let data = [0x01, 0x02, 0x03, 0x04, 0x05, 0x06, 0x07, 0x08];
40//!
41//! j1939.send_message(pgn, dest_addr, &data);
42//!
43//! # Receive J1939 messages
44//! let msg = j1939.receive_message();
45//! ```
46

49use crate::error::Result;
50use crate::types::{Address, Config};

52/// Network layer trait that must be implemented by J1939
53pub trait NetworkLayer: Send + Sync {
54    type Config: Config;
55    type Message;
56
57    fn new(config: Self::Config) -> Result<Self>
58    where
59        Self: Sized;
60    fn open(&mut self) -> Result<()>;
61    fn close(&mut self) -> Result<()>;
62    fn send(&mut self, address: &Address, data: &[u8]) -> Result<()>;
63    fn receive(&mut self) -> Result<Self::Message>;
64    fn set_timeout(&mut self, timeout_ms: u32) -> Result<()>;
65    fn claim_address(&mut self, address: u8) -> Result<()>;
66    fn get_address(&self) -> Result<u8>;
67}
